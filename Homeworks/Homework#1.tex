        \part Construct a predictive equation for the bacteria count Y at time t.
        \begin{solution}
            \begin{verbatim}
                # R code snippet
                # lm_fit <- lm(y_log~t)
                # summary(lm_fit)
            \end{verbatim}

            $ log(\hat{y})  = 5.9732 -0.2184t$\\
